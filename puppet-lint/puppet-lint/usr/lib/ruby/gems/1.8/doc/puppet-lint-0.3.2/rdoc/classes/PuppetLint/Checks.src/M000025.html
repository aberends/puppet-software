  def initialize
    @problems = []
    @default_info = {:check => 'unknown', :linenumber => 0, :column => 0}

    PuppetLint.configuration.checks.each do |check|
      method = PuppetLint.configuration.check_method[check]
      self.class.send(:define_method, "lint_check_#{check}", &method)
    end
  end
