    def initialize(code, offset)
      chunk = code[0..offset]
      @line_no = chunk.count("\n") + 1
      if @line_no == 1
        @column = chunk.length
      else
        @column = chunk.length - chunk.rindex("\n") - 1
      end
      @column = 1 if @column == 0
    end
